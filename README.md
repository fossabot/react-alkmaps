## Install

```bash
npm install --save react-alkmaps
```

## Usage

```jsx
import React, { Component } from "react";

import ReactAlkMaps from "react-alkmaps";

class Example extends Component {
  render() {
    return (
      <ReactAlkMaps
        onLoad={this.onLoad}
        onError={this.onError}
        url="https://maps.alk.com/api/1.2/alkmaps.js"
      />
    );
  }
}
```
